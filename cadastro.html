async function saveEquipe(e) { e.preventDefault(); const getV = (id) => { const el = document.getElementById(id); return el ? el.value.trim() : ''; }; // Validação extra de campos obrigatórios const camposObrigatorios = ['eqp-nome', 'eqp-cpf', 'eqp-telefone', 'eqp-endereco', 'eqp-chave_pix', 'eqp-valor_diaria']; for (let campo of camposObrigatorios) { if (!getV(campo)) { showToast(`Preencha o campo ${campo.replace('eqp-', '').replace('_', ' ')}`, true); return; } } // Tratamento do CPF: remove formatação e valida const cpfFormatado = getV('eqp-cpf'); const cpfNumerico = cpfFormatado.replace(/\D/g, ''); if (cpfNumerico.length !== 11) { showToast('CPF deve conter 11 dígitos numéricos', true); return; } const idAtual = getV('eqp-id'); const isNew = !idAtual; // Verifica duplicidade de CPF no banco (exceto para o próprio registro em edição) showLoading(true); try { let query = sb.from('jsp_equipe').select('id').eq('cpf', cpfNumerico); if (!isNew) { query = query.neq('id', idAtual); } const { data: duplicado, error: errDuplicado } = await query.maybeSingle(); if (errDuplicado) { showLoading(false); showToast('Erro ao verificar CPF: ' + errDuplicado.message, true); return; } if (duplicado) { showLoading(false); showToast('Este CPF já está cadastrado no sistema.', true); return; } } catch (ex) { showLoading(false); showToast('Erro na verificação de duplicidade', true); return; } const payload = { nome: getV('eqp-nome'), categoria: getV('eqp-categoria'), telefone: getV('eqp-telefone'), cpf: cpfNumerico, // Salva apenas números rg: getV('eqp-rg'), endereco: getV('eqp-endereco'), chave_pix: getV('eqp-chave_pix'), obra_atual_id: getV('eqp-obra_atual_id') || null, valor_diaria: parseFloat(getV('eqp-valor_diaria')) || 0, ativo: getV('eqp-ativo') === 'true' }; if(isNew) { payload.id = gerarIdSeguro(); let maxMat = 0; if (STATE.equipe && STATE.equipe.length > 0) { STATE.equipe.forEach(item => { let m = parseInt(item.matricula); if (!isNaN(m) && m > maxMat) maxMat = m; }); } payload.matricula = maxMat > 0 ? maxMat + 1 : 133; payload.contrato_assinado = false; } else { payload.id = idAtual; } const { error } = await sb.from('jsp_equipe').upsert(payload); if(error) { showLoading(false); return showToast("Erro: " + error.message, true); } closeOverlayForm('equipe-form-container'); if(isNew) { const nomeEl = document.getElementById('sucesso-nome'); const matEl = document.getElementById('sucesso-num-matricula'); const modalSucesso = document.getElementById('modal-sucesso-matricula'); if(nomeEl) nomeEl.innerText = payload.nome; if(matEl) matEl.innerText = payload.matricula; if(modalSucesso) modalSucesso.classList.remove('hidden'); lucide.createIcons(); } else { showToast("Funcionário atualizado com sucesso!"); } await loadData(); }
